import Avatar from "@mui/material/Avatar";
import Button from "@mui/material/Button";
import CssBaseline from "@mui/material/CssBaseline";
import TextField from "@mui/material/TextField";
import Link from "@mui/material/Link";

import { createTheme, ThemeProvider } from "@mui/material/styles";
import { useNavigate } from "react-router-dom";
import React, { useState } from "react";
import axios from "axios";

const defaultTheme = createTheme();
var registerSuccess = true;

export default function SignUp() {
  const navigate = useNavigate();
  const [getEmail, setEmail] = useState('');
  const [getPassword, setPassword] = useState('');

  const registerSubmit = (e) => {
    e.preventDefault()
    registerSuccess = true
    if (getEmail === ''){
      alert("Input valid email")
    // }else if (getPassword === ''){
    //   alert("Input valid password")
    } else{
      axios.get('http://127.0.0.1:8000/api/login').then(res => {
        const users = res.data;
        users.map(function(users) {
            if (users.email === getEmail) {
                registerSuccess = false;
            }});
        if (registerSuccess){
            axios.post('http://localhost:8000/api/register',{Email : getEmail, Password : getPassword})
            .then(function (response){alert(response)})
            .catch(function (error) {alert(error)})
        } else {
            alert("Username is already taken"); 
            console.log("Sign-in failed"); 
        }
    })
  }}


  const handleSubmit = (event) => {
    event.preventDefault();
    const data = new FormData(event.currentTarget);
    const email=  data.get('email');
    const password = data.get('password');

    const userInfo = {
        'email' : email,
        'password' : password
    }

    axios.post('http://127.0.0.1:8000/api/register', {
        'email': email,
        'password': password
    }).then((response) => {
        console.log(response);
        navigate('/login')
    }, (error) => {
        console.log(error);
    });
    //navigate('/main-page')
    
};


  return (
    <ThemeProvider theme={defaultTheme}>
      <Container component="main" maxWidth="xs">
        <CssBaseline />
        <Box
          sx={{
            marginTop: 8,
            display: "flex",
            flexDirection: "column",
            alignItems: "center",
          }}
        >
          <Avatar sx={{ m: 1, bgcolor: "secondary.main" }}>
            <LockOutlinedIcon />
          </Avatar>
          <Typography component="h1" variant="h5">
            Sign up
          </Typography>
          <Box
            component="form"
            noValidate
            onSubmit={registerSubmit}
            sx={{ mt: 3 }}
          >
            <Grid container spacing={2}>
              <Grid item xs={12}>
                <TextField
                  required
                  fullWidth
                  id="email"
                  label="Email Address"
                  name="getEmail"

                />
              </Grid>
              <Grid item xs={12}>
                <TextField
                  required
                  fullWidth
                  name="getPassword"
                  label="Password"
                  type="password"
                  id="password"
                  autoComplete="new-password"
                  value={getPassword}
                />
              </Grid>
            </Grid>

              fullWidth

              sx={{ mt: 3, mb: 2 }}
